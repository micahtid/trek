```typescript
export const updateAiMetadata = mutation({
  args: {
    entryId: v.id("entries"),
    skillTags: v.optional(v.array(v.string())),
    impactLevel: v.optional(v.string()),
    primaryCategory: v.optional(v.string()),
    secondaryCategories: v.optional(v.array(v.string())),
    aiDrilldownStatus: v.optional(v.string()),
    aiDrilldownMessages: v.optional(v.array(v.object({
      text: v.string(),
      isUser: v.boolean(),
    }))),
    body: v.optional(v.string()),
  },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) throw new Error("Not authenticated");

    const { entryId, ...fields } = args;
    // Filter out undefined fields so we only patch what's provided
    const patch: Record<string, any> = {};
    for (const [key, value] of Object.entries(fields)) {
      if (value !== undefined) patch[key] = value;
    }
    if (Object.keys(patch).length > 0) {
      await ctx.db.patch(entryId, patch);
    }
  },
});
```

```dart
Future<void> updateAiMetadata({
  required String entryId,

  String? impactLevel,
```
